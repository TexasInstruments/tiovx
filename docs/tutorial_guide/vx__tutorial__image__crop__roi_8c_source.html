125 void vx_tutorial_image_crop_roi()
126 {
134  vx_context context;
135  vx_image image, roi_image;
136  vx_uint32 width, height;
137  vx_rectangle_t rect;
139  void *bmp_file_context;
140 
141  printf(" vx_tutorial_image_crop_roi: Tutorial Started !!! \n");
142 
150  context = vxCreateContext();
153  printf(" Loading file %s ...\n", IN_FILE_NAME);
154 
155  image = load_image_from_handle_from_file(context, IN_FILE_NAME, vx_false_e, &bmp_file_context);
156 
157  vxSetReferenceName((vx_reference)image, "ORIGINAL");
158 
166  show_image_attributes(image);
169  vxQueryImage(image, VX_IMAGE_WIDTH, &width, sizeof(vx_uint32));
170  vxQueryImage(image, VX_IMAGE_HEIGHT, &height, sizeof(vx_uint32));
171 
172  rect.start_x = width/4;
173  rect.start_y = height/4;
174  rect.end_x = rect.start_x + width/2;
175  rect.end_y = rect.start_y + height/2;
176 
183  roi_image = vxCreateImageFromROI(image, &rect);
184 
185  vxSetReferenceName((vx_reference)roi_image, "CROP_ROI");
186 
194  show_image_attributes(roi_image);
197  printf(" Saving to file %s ...\n", OUT_FILE_NAME);
198 
207  save_image_to_file(OUT_FILE_NAME, roi_image);
216  vxReleaseImage(&roi_image);
217  vxReleaseImage(&image);
220  /* now it is safe to release the memory passed to the image
221  * NOTE: "roi_image" also holds a reference to the memory since
222  * its created from the "image"
223  * Hence memory passed to "image" should be freed after "roi_image"
224  * is released
225  */
226  bmp_file_read_release(bmp_file_context);
227 
236  vxReleaseContext(&context);
239  printf(" vx_tutorial_image_crop_roi: Tutorial Done !!! \n");
240  printf(" \n");
241 }

254 vx_image load_image_from_handle_from_file(
255  vx_context context,
256  char *filename,
257  vx_bool convert_to_gray_scale,
258  void **bmp_file_context)
259 {
260  vx_image image = NULL;
261  uint32_t width, height, stride;
262  vx_df_image df;
263  vx_status status;
264  void *data_ptr;
265 
266  status = bmp_file_read(
267  filename,
268  convert_to_gray_scale,
269  &width, &height, &stride, &df, &data_ptr,
270  bmp_file_context);
271 
272  if(status==VX_SUCCESS)
273  {
274  vx_imagepatch_addressing_t image_addr[1];
275  void *ptrs[1];
276  uint32_t bpp;
277 
278  if( df == VX_DF_IMAGE_U8 )
279  bpp = 1;
280  else
281  if( df == VX_DF_IMAGE_RGB )
282  bpp = 3;
283  else
284  if( df == VX_DF_IMAGE_RGBX )
285  bpp = 4;
286  else
287  bpp = 1; /* it should not reach here for BMP files */
288 
289  image_addr[0].dim_x = width;
290  image_addr[0].dim_y = height;
291  image_addr[0].stride_x = bpp;
292  image_addr[0].stride_y = stride;
293  image_addr[0].scale_x = VX_SCALE_UNITY;
294  image_addr[0].scale_y = VX_SCALE_UNITY;
295  image_addr[0].step_x = 1;
296  image_addr[0].step_y = 1;
297 
298  ptrs[0] = data_ptr;
299 
306  image = vxCreateImageFromHandle(context, df, image_addr, ptrs, VX_MEMORY_TYPE_HOST);
309  /* MUST not free memory that is passed to vxCreateImageFromHandle, until
310  * that memory ptr is swapped out from the image
311  * or the image handle is released
312  */
313  /* bmp_file_read_release(bmp_file_context); */
314  }
315  return image;
316 }
