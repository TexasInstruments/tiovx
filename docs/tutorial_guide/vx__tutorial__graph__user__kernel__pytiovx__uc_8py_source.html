00084 class NodePhaseRgb (Node) :
00085     ## Constructor for user/target kernel class
00086     #
00087     #  \param image_in [in] Input image handle parameter for this function
00088     #  \param image_out [in] Output image handle parameter for this function
00089     #  \param name [in] user indetifiable name
00090     #  \param target [in] target CPU on which this kernel/node will run
00091     def __init__(self, image_in, image_out, name="default", target=Target.DEFAULT) :
00092         Node.__init__(self, "vx_tutorial_graph.phase_rgb", image_in, image_out)
00093         self.setParams(1, 1, Type.IMAGE, Type.IMAGE)
00094         self.setTarget(target)
00095         self.setKernelEnumName("VX_USER_KERNEL")

00097     ## Parameter checking function
00098     #
00099     #  Checks if number of parameters passed is correct
00100     #  Checks if data type of parameters passed is correct
00101     #
00102     #  \param param_type_args [in] Variable number of args list of type tiovx.enums.Type to specify data object types
00103     def checkParams(self, *param_type_args) :
00104         # invoke base class checker function
00105         Node.checkParams(self, *param_type_args)
00106         # additional error conditions over the basic ones
00107         assert ( self.ref[0].width    == self.ref[1].width ), "Input and Output width MUST match"
00108         assert ( self.ref[0].height   == self.ref[1].height ), "Input and Output height MUST match"
00109         assert ( self.ref[0].df_image == DfImage.U8 ), "Input data format must be U8"
00110         assert ( self.ref[1].df_image == DfImage.RGB ), "Output data format must be RGB"

00113 def make_my_graph() :

00136

00139 
00140     ## Create and add node to graph for Sobel3x3
00141     graph.add ( NodeSobel3x3(in_image, grad_x, grad_y, target=Target.DSP1) )

00143     ## Create and add node to graph for Phase
00144     graph.add ( NodePhase(grad_x, grad_y, phase, target=Target.DSP1) )
00145 
00146     ## Create and add node to graph for user/target kernel
00147     graph.add ( NodePhaseRgb(phase, phase_rgb, target=Target.DSP1) )
00148 
00149     ## Add graph to context
00150     context.add ( graph )
