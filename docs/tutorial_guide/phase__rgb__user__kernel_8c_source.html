415 static vx_status VX_CALLBACK phase_rgb_user_kernel_validate(vx_node node,
416  const vx_reference parameters[ ],
417  vx_uint32 num,
418  vx_meta_format metas[])
419 {
420  vx_status status = VX_SUCCESS;
421  vx_image img[PHASE_RGB_MAX_PARAMS];
422  vx_uint32 w[PHASE_RGB_MAX_PARAMS];
423  vx_uint32 h[PHASE_RGB_MAX_PARAMS];
424  vx_uint32 out_w, out_h, i;
425  vx_df_image fmt[PHASE_RGB_MAX_PARAMS], out_fmt;
426 
427  if (num != PHASE_RGB_MAX_PARAMS)
428  {
429  printf(" phase_rgb_user_kernel_validate: ERROR: Number of parameters dont match !!!\n");
430  status = VX_ERROR_INVALID_PARAMETERS;
431  }
432 
433  for (i = 0U; i < PHASE_RGB_MAX_PARAMS; i ++)
434  {
435  img[i] = (vx_image)parameters[i];
436 
437  /* Check for NULL */
438  if (NULL == img[i])
439  {
440  printf(" phase_rgb_user_kernel_validate: ERROR: Parameter %d is NULL !!!\n", i);
441  status = VX_ERROR_NO_MEMORY;
442  break;
443  }
444  }
445  if (VX_SUCCESS == status)
446  {
447  /* Get the image width/heigh and format */
448  status = vxQueryImage(img[PHASE_RGB_IN0_IMG_IDX],
449  VX_IMAGE_FORMAT, &fmt[PHASE_RGB_IN0_IMG_IDX],
450  sizeof(vx_df_image));
451  status |= vxQueryImage(img[PHASE_RGB_IN0_IMG_IDX],
452  VX_IMAGE_WIDTH, &w[PHASE_RGB_IN0_IMG_IDX],
453  sizeof(vx_uint32));
454  status |= vxQueryImage(img[PHASE_RGB_IN0_IMG_IDX],
455  VX_IMAGE_HEIGHT, &h[PHASE_RGB_IN0_IMG_IDX],
456  sizeof(vx_uint32));
457 
458  if(status!=VX_SUCCESS)
459  {
460  printf(" phase_rgb_user_kernel_validate: ERROR: Unable to query input image !!!\n");
461  }
462  }
463  if ((VX_SUCCESS == status) &&
464  (vx_false_e == tivxIsReferenceVirtual(
465  parameters[PHASE_RGB_OUT0_IMG_IDX])))
466  {
467  /* Get the image width/heigh and format */
468  status = vxQueryImage(img[PHASE_RGB_OUT0_IMG_IDX],
469  VX_IMAGE_FORMAT, &fmt[PHASE_RGB_OUT0_IMG_IDX],
470  sizeof(vx_df_image));
471  status |= vxQueryImage(img[PHASE_RGB_OUT0_IMG_IDX],
472  VX_IMAGE_WIDTH, &w[PHASE_RGB_OUT0_IMG_IDX],
473  sizeof(vx_uint32));
474  status |= vxQueryImage(img[PHASE_RGB_OUT0_IMG_IDX],
475  VX_IMAGE_HEIGHT, &h[PHASE_RGB_OUT0_IMG_IDX],
476  sizeof(vx_uint32));
477 
478  if(status!=VX_SUCCESS)
479  {
480  printf(" phase_rgb_user_kernel_validate: ERROR: Unable to query output image !!!\n");
481  }
482  }
483 
484  if (VX_SUCCESS == status)
485  {
486  /* Check for validity of data format */
487  if (VX_DF_IMAGE_U8 != fmt[PHASE_RGB_IN0_IMG_IDX] &&
488  VX_DF_IMAGE_RGB != fmt[PHASE_RGB_OUT0_IMG_IDX]
489  )
490  {
491  status = VX_ERROR_INVALID_PARAMETERS;
492  printf(" phase_rgb_user_kernel_validate: ERROR: Input/Output image format not correct !!!\n");
493  }
494 
495  /* Check for frame sizes */
496  if (vx_false_e == tivxIsReferenceVirtual(
497  parameters[PHASE_RGB_OUT0_IMG_IDX]))
498  {
499  if ((w[PHASE_RGB_IN0_IMG_IDX] !=
500  w[PHASE_RGB_OUT0_IMG_IDX]) ||
501  (h[PHASE_RGB_IN0_IMG_IDX] !=
502  h[PHASE_RGB_OUT0_IMG_IDX]))
503  {
504  status = VX_ERROR_INVALID_PARAMETERS;
505  printf(" phase_rgb_user_kernel_validate: ERROR: Input/Output image wxh do not match !!!\n");
506  }
507  }
508  }
509 
510  if (VX_SUCCESS == status)
511  {
512  out_fmt = VX_DF_IMAGE_RGB;
513  out_w = w[PHASE_RGB_OUT0_IMG_IDX];
514  out_h = h[PHASE_RGB_OUT0_IMG_IDX];
515 
516  vxSetMetaFormatAttribute(metas[PHASE_RGB_OUT0_IMG_IDX], VX_IMAGE_FORMAT, &out_fmt,
517  sizeof(out_fmt));
518  vxSetMetaFormatAttribute(metas[PHASE_RGB_OUT0_IMG_IDX], VX_IMAGE_WIDTH, &out_w,
519  sizeof(out_w));
520  vxSetMetaFormatAttribute(metas[PHASE_RGB_OUT0_IMG_IDX], VX_IMAGE_HEIGHT, &out_h,
521  sizeof(out_h));
522  }
523 
524  if(status==VX_SUCCESS)
525  {
526  printf(" phase_rgb_user_kernel_validate: SUCCESS !!!\n");
527  }
528 
529  return status;
530 }

566 static vx_status VX_CALLBACK phase_rgb_user_kernel_run(vx_node node,
567  const vx_reference parameters[ ],
568  vx_uint32 num)
569 {
570  vx_image in_image = (vx_image)parameters[PHASE_RGB_IN0_IMG_IDX];
571  vx_image out_image = (vx_image)parameters[PHASE_RGB_OUT0_IMG_IDX];
572  vx_rectangle_t rect = { 0 };
573  vx_map_id in_map_id = (vx_map_id)VX_ERROR_INVALID_PARAMETERS, out_map_id = (vx_map_id)VX_ERROR_INVALID_PARAMETERS;
574  vx_imagepatch_addressing_t in_image_addr, out_image_addr;
575  vx_uint32 out_w=0, out_h=0;
576  vx_status status=0;
577  void *in_data_ptr, *out_data_ptr;
578 
579  status |= vxQueryImage(out_image,
580  VX_IMAGE_WIDTH, &out_w,
581  sizeof(vx_uint32));
582  status |= vxQueryImage(out_image,
583  VX_IMAGE_HEIGHT, &out_h,
584  sizeof(vx_uint32));
585 
586  rect.start_x = 0;
587  rect.start_y = 0;
588  rect.end_x = out_w;
589  rect.end_y = out_h;
590 
591  if(status==VX_SUCCESS)
592  {
593  status = vxMapImagePatch(in_image,
594  &rect,
595  0,
596  &in_map_id,
597  &in_image_addr,
598  (void**)&in_data_ptr,
599  VX_READ_ONLY, VX_MEMORY_TYPE_HOST, VX_NOGAP_X);
600  }
601 
602  if(status==VX_SUCCESS)
603  {
604  status = vxMapImagePatch(out_image,
605  &rect,
606  0,
607  &out_map_id,
608  &out_image_addr,
609  (void**)&out_data_ptr,
610  VX_WRITE_ONLY, VX_MEMORY_TYPE_HOST, VX_NOGAP_X);
611  }
612  if(status==VX_SUCCESS)
613  {
614  vx_uint32 x, y;
615 
616  for(y=0; y<out_image_addr.dim_y; y++)
617  {
618  for(x=0; x<out_image_addr.dim_x; x++)
619  {
620  vx_uint8 *in_pixel = (vx_uint8 *)vxFormatImagePatchAddress2d(in_data_ptr, x, y, &in_image_addr);
621  vx_uint8 *out_pixel = (vx_uint8 *)vxFormatImagePatchAddress2d(out_data_ptr, x, y, &out_image_addr);
622 
623  if(*in_pixel<64)
624  {
625  out_pixel[0] = 0xFF;
626  out_pixel[1] = 0x00;
627  out_pixel[2] = 0x00;
628  }
629  else
630  if(*in_pixel<128)
631  {
632  out_pixel[0] = 0x00;
633  out_pixel[1] = 0xFF;
634  out_pixel[2] = 0x00;
635  }
636  else
637  if(*in_pixel<192)
638  {
639  out_pixel[0] = 0x00;
640  out_pixel[1] = 0x00;
641  out_pixel[2] = 0xFF;
642  }
643  else
644  {
645  out_pixel[0] = 0xFF;
646  out_pixel[1] = 0xFF;
647  out_pixel[2] = 0xFF;
648  }
649  }
650  }
651 
652  }
653 
654  if(in_map_id!=(vx_map_id)VX_ERROR_INVALID_PARAMETERS)
655  {
656  vxUnmapImagePatch(in_image, in_map_id);
657  }
658  if(out_map_id!=(vx_map_id)VX_ERROR_INVALID_PARAMETERS)
659  {
660  vxUnmapImagePatch(out_image, out_map_id);
661  }
662 
663  if(status!=VX_SUCCESS)
664  {
665  printf(" phase_rgb_user_kernel_run: ERROR: Run failed (%d)!!!\n", status);
666  }
667  return status;
668 }
