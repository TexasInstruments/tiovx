00360 static vx_status VX_CALLBACK phase_rgb_user_kernel_run(vx_node node,
00361             const vx_reference parameters[ ],
00362             vx_uint32 num)
00363 {
00364     vx_image in_image = (vx_image)parameters[PHASE_RGB_IN0_IMG_IDX];
00365     vx_image out_image = (vx_image)parameters[PHASE_RGB_OUT0_IMG_IDX];
00366     vx_rectangle_t rect = { 0 };
00367     vx_map_id in_map_id = (vx_map_id)VX_ERROR_INVALID_PARAMETERS, out_map_id = (vx_map_id)VX_ERROR_INVALID_PARAMETERS;
00368     vx_imagepatch_addressing_t in_image_addr, out_image_addr;
00369     vx_uint32 out_w=0, out_h=0;
00370     vx_status status=0;
00371     void *in_data_ptr, *out_data_ptr;
00372

00374         VX_IMAGE_WIDTH, &out_w,
00375         sizeof(vx_uint32));
00376     status |= vxQueryImage(out_image,
00377         VX_IMAGE_HEIGHT, &out_h,
00378         sizeof(vx_uint32));
00379 
00380     rect.start_x = 0;
00381     rect.start_y = 0;
00382     rect.end_x   = out_w;
00383     rect.end_y   = out_h;
00384

00386     {
00387         status = vxMapImagePatch(in_image,
00388                 &rect,
00389                 0,
00390                 &in_map_id,
00391                 &in_image_addr,
00392                 (void**)&in_data_ptr,

00395

00398         status = vxMapImagePatch(out_image,
00399                 &rect,
00400                 0,
00401                 &out_map_id,
00402                 &out_image_addr,
00403                 (void**)&out_data_ptr,
00404                 VX_WRITE_ONLY, VX_MEMORY_TYPE_HOST, VX_NOGAP_X);

00407     {
00408         vx_uint32 x, y;
00409 
00410         for(y=0; y<out_image_addr.dim_y; y++)
00411         {
00412             for(x=0; x<out_image_addr.dim_x; x++)
00413             {
00414                 vx_uint8 *in_pixel = (vx_uint8 *)vxFormatImagePatchAddress2d(in_data_ptr, x, y, &in_image_addr);
00415                 vx_uint8 *out_pixel = (vx_uint8 *)vxFormatImagePatchAddress2d(out_data_ptr, x, y, &out_image_addr);
00416 
00417                 if(*in_pixel<64)
00418                 {
00419                     out_pixel[0] = 0xFF;

00425                 {
00426                     out_pixel[0] = 0x00;
00427                     out_pixel[1] = 0xFF;
00428                     out_pixel[2] = 0x00;
00429                 }
00430                 else
00431                 if(*in_pixel<192)
00432                 {
00433                     out_pixel[0] = 0x00;
00434                     out_pixel[1] = 0x00;
00435                     out_pixel[2] = 0xFF;
00436                 }
00437                 else
00438                 {
00439                     out_pixel[0] = 0xFF;
00440                     out_pixel[1] = 0xFF;
00441                     out_pixel[2] = 0xFF;
00442                 }
00443             }
00444         }
