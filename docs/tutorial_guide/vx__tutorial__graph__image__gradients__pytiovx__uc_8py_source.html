63 ## \file vx_tutorial_graph_image_gradients_pytiovx_uc.py This file show example of using basic PyTIOVX APIs to
64 # describe an OpenVX graph.
65 #
66 # To import PyTIOVX module so that the APIs can be called in this file add below line
67 # \code
68 # from tiovx import *
69 # \endcode
70 #
71 # The basic steps one follows to describe the graph for PyTIOVX tool are,
72 # - Create a context object.
73 # - Pass as input a string which is used as a perfix to name the generated code and function.
74 # - Create a graph object object
75 # - Create data objects.
76 # - TIP: specify a name for data objects via name="xyz" to be able to identify it in the generated
77 # C code and image file
78 # - Create node objects and link it to required data objects
79 # - TIP: Add nodes to graph as you create them using graph.add()
80 # - TIP: Use target=Target.<TargetName> to specify the target on which this node runs.
81 # - Add graph to context using context.add()
82 # - Generate C code for the objects in a context using
83 # \code
84 # ExportCode(context).export()
85 # \endcode
86 # - Generate image file for the objects in a context using
87 # \code
88 # ExportImage(context).export()
89 # \endcode
90 # - Run the PyTIOVX on this file using
91 # \code
92 # python <.py file>
93 # \endcode
94 # - In case of any error reported by the PYTIOVX tool, correct the input .py file
95 # - Verify the generated image file to confirm the graph is as expected.
96 # - Write the remaining C code to invoke the generated APIs
97 #
98 # See source code of function vx_tutorial_graph_image_gradients_pytiovx_uc.make_my_graph() for detailed API usage for this example.

102 ## Function to describe a graph and generate code, image using PyTIOVX tool
103 def make_my_graph() :
104  ## Create a context object.
105  context = Context("vx_tutorial_graph_image_gradients_pytiovx_uc")
106 
107  ## Create a graph object.
108  graph = Graph()
109 
110  ## local variables to define the with and height of input image
111  width = 640
112  ## local variables to define the with and height of input image
113  height = 480
114 
115  ## Create input image object.
116  in_image = Image(width, height, DfImage.U8, name="input")
117 
118  ## Create grad_x image object.
119  grad_x = Image(width, height, DfImage.S16, name="grad_x")
120 
121  ## Create grad_y image object.
122  grad_y = Image(width, height, DfImage.S16, name="grad_y")
123 
124  ## Create magnitude image object.
125  magnitude = Image(width, height, DfImage.S16, name="magnitude")
126 
127  ## Create phase image object.
128  phase = Image(width, height, DfImage.U8, name="phase")
129 
130  ## Create image object to represent grad_x as grayscale image
131  grad_x_img = Image(width, height, DfImage.U8, name="grad_x_img")
132 
133  ## Create image object to represent grad_y as grayscale image
134  grad_y_img = Image(width, height, DfImage.U8, name="grad_y_img")
135 
136  ## Create image object to represent magnitude as grayscale image
137  magnitude_img = Image(width, height, DfImage.U8, name="magnitude_img")
138 
139  ## Create a scalar object to represent 'shift' for ConvertDepth
140  shift = Scalar(Type.INT32, 0, name="shift")
141 
142  ## Create and add node to graph for Sobel3x3
143  graph.add ( NodeSobel3x3(in_image, grad_x, grad_y, target=Target.DSP1) )
144 
145  ## Create and add node to graph for Magnitude
146  graph.add ( NodeMagnitude(grad_x, grad_y, magnitude, target=Target.DSP2) )
147 
148  ## Create and add node to graph for Phase
149  graph.add ( NodePhase(grad_x, grad_y, phase, target=Target.DSP1) )
150 
151  ## Create and add node to graph for ConvertDepth to convert magnitude to 8b grayscale image
152  graph.add ( NodeConvertDepth(magnitude, magnitude_img, Policy.SATURATE, shift, target=Target.DSP2) )
153 
154  ## Create and add node to graph for ConvertDepth to convert grad_x to 8b grayscale image
155  graph.add ( NodeConvertDepth(grad_x, grad_x_img, Policy.SATURATE, shift, target=Target.DSP2) )
156 
157  ## Create and add node to graph for ConvertDepth to convert grad_y to 8b grayscale image
158  graph.add ( NodeConvertDepth(grad_y, grad_y_img, Policy.SATURATE, shift, target=Target.DSP1) )
159 
160  ## Add graph to context
161  context.add ( graph )
162 
163  ## Generate .jpg using 'dot' tool to generate visual representation of graphs in this context
164  ExportImage(context).export()
165  ## Generate C code to create, run, delete graphs in this context
166  ExportCode(context).export()
