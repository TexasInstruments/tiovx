359 {
360  vx_uint32 width, height;

362  vx_rectangle_t rect;

364  vx_df_image df;
365  void *data_ptr;
366  vx_status status;
367 
372  status = vxGetStatus((vx_reference)image);

408  &data_ptr,

411  VX_NOGAP_X
412  );
415  if(status==VX_SUCCESS)
416  {
424  bmp_file_write(filename, width, height, image_addr.stride_y, df, data_ptr);
434  vxUnmapImagePatch(image, map_id);

532 void ascii_file_read(char *filename, int num_elements, void* buffer, vx_enum data_type){
533  FILE* ptr_file;
534 
535  // Initialize buffer for reading each line of file
536  int nbytes = sizeof(float);
537  char* buff = malloc(nbytes);
538 
539  // Initialize variables for tokenizing each line based on delimiter values
540  char* token;
541  char* delim = " \n\t,";
542  float val;
543 
544  // Initialize different data type arrays
545  vx_uint8* u8;
546  vx_int32* i32;
547  vx_float32* f32;
548 
549  // Try reading in file
550  ptr_file = fopen(filename, "r");
551  if(!ptr_file){
552  printf("FAILED READING FILE");
553  }
554 
555  // If file read is successful, try to populate matrix
556  int i = 0;
557  while(fgets(buff, nbytes, ptr_file) > 0){
558  // Split string into tokens at whitespaces, commas, newlines, or tabs
559  for(token = strtok(buff, delim); token != NULL; token = strtok(NULL, delim)){
560  val = strtod(token, NULL);
561 
562  switch(data_type){
563  case VX_TYPE_FLOAT32:
564  f32 = buffer;
565  f32[i] = val;
566  break;
567 
568  case VX_TYPE_UINT8:
569  u8 = buffer;
570  uint8_t val_u8 = val;
571  u8[i] = val_u8;
572  break;
573 
574  case VX_TYPE_INT32:
575  i32 = buffer;
576  int32_t val_int32 = val;
577  i32[i] = val_int32;
578  break;
579 
580  default:
581  break;
582  }
583 
584  i++;
585  if (i >= num_elements){
586  break;
587  }
588  }
589  // Need to break out of both loops when this condition is met
590  if (i >= num_elements){
591  break;
592  }
593  }
594  fclose(ptr_file);
595  //printf("val of first element: %d\n", u8[0]);
596 }
