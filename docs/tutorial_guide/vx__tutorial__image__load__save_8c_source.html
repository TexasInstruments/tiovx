233 vx_image create_image_from_file(vx_context context, char *filename, vx_bool convert_to_gray_scale)
234 {
235  vx_image image = NULL;
236  uint32_t width, height, stride;
237  vx_df_image df;
238  vx_status status;
239  void *data_ptr, *bmp_file_context;
240 
251  #ifdef SKIP_FILEIO
252  status = 0;
253  width = 640;
254  height = 480;
255  stride = 640;
256  if(convert_to_gray_scale)
257  df = VX_DF_IMAGE_U8;

260  data_ptr = NULL;
261  bmp_file_context = NULL;

286  image = vxCreateImage(context, width, height, df);
287  status = vxGetStatus((vx_reference)image);

290  {
291 
313  vx_imagepatch_addressing_t image_addr;

316 
317  if( df == VX_DF_IMAGE_U8 )

570 void ascii_file_read(char *filename, int num_elements, void* buffer, vx_enum data_type){
571  FILE* ptr_file;
572 
573  // Initialize buffer for reading each line of file
574  int nbytes = sizeof(float);
575  char* buff = malloc(nbytes);
576 
577  // Initialize variables for tokenizing each line based on delimiter values
578  char* token;
579  char* delim = " \n\t,";
580  float val;
581 
582  // Initialize different data type arrays
583  vx_uint8* u8;
584  vx_int32* i32;
585  vx_float32* f32;
586  uint8_t val_u8;
587  int32_t val_int32;
588 
589  // Try reading in file
590  ptr_file = fopen(filename, "r");
591  if(!ptr_file){
592  printf("FAILED READING FILE");
593  }
594 
595  // If file read is successful, try to populate matrix
596  int i = 0;
597  while(fgets(buff, nbytes, ptr_file) > 0){
598  // Split string into tokens at whitespaces, commas, newlines, or tabs
599  for(token = strtok(buff, delim); token != NULL; token = strtok(NULL, delim)){
600  val = strtod(token, NULL);
601 
602  switch(data_type){
603  case VX_TYPE_FLOAT32:
604  f32 = buffer;
605  f32[i] = val;
606  break;
607 
608  case VX_TYPE_UINT8:
609  u8 = buffer;
610  val_u8 = val;
611  u8[i] = val_u8;
612  break;
613 
614  case VX_TYPE_INT32:
615  i32 = buffer;
616  val_int32 = val;
617  i32[i] = val_int32;
618  break;
619 
620  default:
621  break;
622  }
623 
624  i++;
625  if (i >= num_elements){
626  break;
627  }
628  }
629  // Need to break out of both loops when this condition is met
630  if (i >= num_elements){
631  break;
632  }
633  }
634  fclose(ptr_file);
635  //printf("val of first element: %d\n", u8[0]);
636 }
