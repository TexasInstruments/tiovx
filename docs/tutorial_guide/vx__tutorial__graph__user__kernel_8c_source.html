192  context = vxCreateContext();
210  status = phase_rgb_user_kernel_add(context, add_as_target_kernel);

235  vxQueryImage(in_image, VX_IMAGE_HEIGHT, &height, sizeof(vx_uint32));

246  show_image_attributes(out_image);

272  status = vxVerifyGraph(graph);
273 
274  show_graph_attributes(graph);

280  tivxExportGraphToDot(graph, ".", "vx_tutorial_graph_user_kernel");

288  vxWaitGraph(graph);
289 
290  printf(" Executing graph ... Done !!!\n");
291 
292  show_graph_attributes(graph);
293  show_node_attributes(node);
294 
295  printf(" Saving to file %s ...\n", out_file);
296  save_image_to_file(out_file, out_image);
297  }
305  vxReleaseImage(&in_image);
306  vxReleaseImage(&out_image);
307  vxReleaseNode(&node);
308  vxReleaseGraph(&graph);
317  phase_rgb_user_kernel_remove(context);
319  }
325  vxReleaseContext(&context);
328  printf(" vx_tutorial_graph_user_kernel: Tutorial Done !!! \n");
329  printf(" \n");
330 }
