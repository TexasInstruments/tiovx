00137             {

00141             /* scale to image height */
00142             for(i=0; i<num_bins; i++)
00143             {
00144                 histogram_data[i] = (uint32_t)(((float)histogram_data[i]/max)*height + 0.5);
00145                 if(histogram_data[i]>height)
00146                     histogram_data[i] = height;
00147             }
00148 
00149             {
00150                 uint8_t *data_ptr = NULL;
00151                 vx_rectangle_t rect = { 0, 0, width, height};
00152                 vx_map_id map_id;

00154 
00155                 status = vxMapImagePatch(image,
00156                             &rect,
00157                             0,
00158                             &map_id,
00159                             &image_addr,

00161                             VX_WRITE_ONLY, VX_MEMORY_TYPE_HOST, VX_NOGAP_X);
00162

00164                 {

00166                     uint8_t *addr0 = NULL, *addr1 = NULL;

00172                         addr0 = data_ptr + i*bin_width;

00176                             addr1 = addr0 + j*image_addr.stride_y;
