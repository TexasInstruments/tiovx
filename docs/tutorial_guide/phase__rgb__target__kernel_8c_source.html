00035     else

00053             dst_desc->mem_size[0], dst_desc->mem_ptr[0].mem_type,
00054             VX_WRITE_ONLY);

00057         {
00058             vx_uint32 x, y, width, height;
00059             vx_uint8 *in_data_ptr, *out_data_ptr;
00060             vx_uint8 *in_pixel, *out_pixel;

00063             out_data_ptr = dst_desc->mem_ptr[0].target_ptr;
00064

00066             height = dst_desc->imagepatch_addr[0U].dim_y;
00067 
00068             for(y=0; y<height; y++)

00083                     {
00084                         out_pixel[0] = 0x00;
00085                         out_pixel[1] = 0xFF;

00090                     {
