117 vx_status VX_CALLBACK vxTutotrialPhaseRgb(
118  tivx_target_kernel_instance kernel, tivx_obj_desc_t *obj_desc[],
119  uint16_t num_params, void *priv_arg)
120 {
121  vx_status status = VX_SUCCESS;
122  tivx_obj_desc_image_t *src_desc, *dst_desc;
123 
124  if ((num_params != PHASE_RGB_MAX_PARAMS)
125  || (NULL == obj_desc[PHASE_RGB_IN0_IMG_IDX])
126  || (NULL == obj_desc[PHASE_RGB_OUT0_IMG_IDX])
127  )
128  {
129  status = VX_FAILURE;
130  }
131  else
132  {
133  void *src_desc_target_ptr;
134  void *dst_desc_target_ptr;
135 
136  /* Get the Src and Dst descriptors */
137  src_desc = (tivx_obj_desc_image_t *)obj_desc[PHASE_RGB_IN0_IMG_IDX];
138  dst_desc = (tivx_obj_desc_image_t *)obj_desc[PHASE_RGB_OUT0_IMG_IDX];
139 
140  /* Get the target pointer from the shared pointer for all
141  buffers */
142  src_desc_target_ptr = tivxMemShared2TargetPtr(
143  src_desc->mem_ptr[0].shared_ptr, src_desc->mem_ptr[0].mem_heap_region);
144  dst_desc_target_ptr = tivxMemShared2TargetPtr(
145  dst_desc->mem_ptr[0].shared_ptr, dst_desc->mem_ptr[0].mem_heap_region);
146 
147  /* Map all buffers, which invalidates the cache */
148  tivxMemBufferMap(src_desc_target_ptr,
149  src_desc->mem_size[0], VX_MEMORY_TYPE_HOST,
150  VX_READ_ONLY);
151  tivxMemBufferMap(dst_desc_target_ptr,
152  dst_desc->mem_size[0], VX_MEMORY_TYPE_HOST,
153  VX_WRITE_ONLY);
154 
155  /* Run function for complete image, ROI is ignored in this example */
156  {
157  vx_uint32 x, y, width, height;
158  vx_uint8 *in_data_ptr, *out_data_ptr;
159  vx_uint8 *in_pixel, *out_pixel;
160 
161  in_data_ptr = src_desc_target_ptr;
162  out_data_ptr = dst_desc_target_ptr;
163 
164  width = dst_desc->imagepatch_addr[0U].dim_x;
165  height = dst_desc->imagepatch_addr[0U].dim_y;
166 
167  for(y=0; y<height; y++)
168  {
169  in_pixel = in_data_ptr;
170  out_pixel = out_data_ptr;
171 
172  for(x=0; x<width; x++)
173  {
174  if(in_pixel[0]<64)
175  {
176  out_pixel[0] = 0xFF;
177  out_pixel[1] = 0x00;
178  out_pixel[2] = 0x00;
179  }
180  else
181  if(in_pixel[0]<128)
182  {
183  out_pixel[0] = 0x00;
184  out_pixel[1] = 0xFF;
185  out_pixel[2] = 0x00;
186  }
187  else
188  if(in_pixel[0]<192)
189  {
190  out_pixel[0] = 0x00;
191  out_pixel[1] = 0x00;
192  out_pixel[2] = 0xFF;
193  }
194  else
195  {
196  out_pixel[0] = 0xFF;
197  out_pixel[1] = 0xFF;
198  out_pixel[2] = 0xFF;
199  }
200  in_pixel++;
201  out_pixel += 3;
202  }
203  in_data_ptr += src_desc->imagepatch_addr[0U].stride_y;
204  out_data_ptr += dst_desc->imagepatch_addr[0U].stride_y;
205  }
206  }
207 
208  tivxMemBufferUnmap(src_desc_target_ptr,
209  src_desc->mem_size[0], VX_MEMORY_TYPE_HOST,
210  VX_READ_ONLY);
211  tivxMemBufferUnmap(dst_desc_target_ptr,
212  dst_desc->mem_size[0], VX_MEMORY_TYPE_HOST,
213  VX_WRITE_ONLY);
214 
215  }
216 
217  return (status);
218 }
