00117 vx_status VX_CALLBACK vxTutotrialPhaseRgb(
00118     tivx_target_kernel_instance kernel, tivx_obj_desc_t *obj_desc[],
00119     uint16_t num_params, void *priv_arg)

00121     vx_status status = VX_SUCCESS;
00122     tivx_obj_desc_image_t *src_desc, *dst_desc;
00123 
00124     if ((num_params != PHASE_RGB_MAX_PARAMS)
00125         || (NULL == obj_desc[PHASE_RGB_IN0_IMG_IDX])
00126         || (NULL == obj_desc[PHASE_RGB_OUT0_IMG_IDX])
00127         )
00128     {
00129         status = VX_FAILURE;
00130     }
00131     else
00132     {
00133         /* Get the Src and Dst descriptors */
00134         src_desc = (tivx_obj_desc_image_t *)obj_desc[PHASE_RGB_IN0_IMG_IDX];
00135         dst_desc = (tivx_obj_desc_image_t *)obj_desc[PHASE_RGB_OUT0_IMG_IDX];
00136 
00137         /* Get the target pointer from the shared pointer for all
00138            buffers */
00139         src_desc->mem_ptr[0].target_ptr = tivxMemShared2TargetPtr(
00140             src_desc->mem_ptr[0].shared_ptr, src_desc->mem_ptr[0].mem_type);
00141         dst_desc->mem_ptr[0].target_ptr = tivxMemShared2TargetPtr(
00142             dst_desc->mem_ptr[0].shared_ptr, dst_desc->mem_ptr[0].mem_type);
00143 
00144         /* Map all buffers, which invalidates the cache */
00145         tivxMemBufferMap(src_desc->mem_ptr[0].target_ptr,
00146             src_desc->mem_size[0], src_desc->mem_ptr[0].mem_type,
00147             VX_READ_ONLY);
00148         tivxMemBufferMap(dst_desc->mem_ptr[0].target_ptr,
00149             dst_desc->mem_size[0], dst_desc->mem_ptr[0].mem_type,
00150             VX_WRITE_ONLY);
00151 
00152         /* Run function for complete image, ROI is ignored in this example */
00153         {
00154             vx_uint32 x, y, width, height;
00155             vx_uint8 *in_data_ptr, *out_data_ptr;
00156             vx_uint8 *in_pixel, *out_pixel;
00157 
00158             in_data_ptr = src_desc->mem_ptr[0].target_ptr;
00159             out_data_ptr = dst_desc->mem_ptr[0].target_ptr;
00160 
00161             width = dst_desc->imagepatch_addr[0U].dim_x;
00162             height = dst_desc->imagepatch_addr[0U].dim_y;
00163 
00164             for(y=0; y<height; y++)
00165             {
00166                 in_pixel = in_data_ptr;
00167                 out_pixel = out_data_ptr;
00168 
00169                 for(x=0; x<width; x++)
00170                 {
00171                     if(in_pixel[0]<64)
00172                     {
00173                         out_pixel[0] = 0xFF;
00174                         out_pixel[1] = 0x00;
00175                         out_pixel[2] = 0x00;
00176                     }
00177                     else
00178                     if(in_pixel[0]<128)
00179                     {
00180                         out_pixel[0] = 0x00;
00181                         out_pixel[1] = 0xFF;
00182                         out_pixel[2] = 0x00;
00183                     }
00184                     else
00185                     if(in_pixel[0]<192)
00186                     {
00187                         out_pixel[0] = 0x00;
00188                         out_pixel[1] = 0x00;
00189                         out_pixel[2] = 0xFF;
00190                     }
00191                     else
00192                     {
00193                         out_pixel[0] = 0xFF;
00194                         out_pixel[1] = 0xFF;
00195                         out_pixel[2] = 0xFF;
00196                     }
00197                     in_pixel++;
00198                     out_pixel += 3;
00199                 }
00200                 in_data_ptr += src_desc->imagepatch_addr[0U].stride_y;
00201                 out_data_ptr += dst_desc->imagepatch_addr[0U].stride_y;
00202             }
00203         }
00204 
00205         tivxMemBufferUnmap(src_desc->mem_ptr[0].target_ptr,
00206             src_desc->mem_size[0], src_desc->mem_ptr[0].mem_type,
00207             VX_READ_ONLY);
00208         tivxMemBufferUnmap(dst_desc->mem_ptr[0].target_ptr,
00209             dst_desc->mem_size[0], dst_desc->mem_ptr[0].mem_type,
00210             VX_WRITE_ONLY);
00211 
00212     }
00213 
00214     return (status);
