135 void vx_tutorial_graph_pipeline_two_nodes()
136 {
137 /* max number of buffers to allocate for input and output */
138 #define MAX_NUM_BUF (2u)
139 
151  vx_context context;
152  vx_image in_img[MAX_NUM_BUF], tmp, out_img[MAX_NUM_BUF];
153  vx_node n0, n1;
154  vx_graph_parameter_queue_params_t graph_parameters_queue_params_list[2];
155  vx_graph graph;
156  uint32_t width, height, num_buf, pipeline_depth, buf_id, loop_id, loop_cnt, exe_time;
159  printf(" vx_tutorial_graph_pipeline_two_nodes: Tutorial Started !!! \n");
160 
168  context = vxCreateContext();
171  width = 1000;
172  height = 1000;
173  num_buf = MAX_NUM_BUF;
174  pipeline_depth = MAX_NUM_BUF;
175  loop_cnt = 10 - num_buf; /* runs the graph 10 times */
176 
177  graph = vxCreateGraph(context);
178 
186  for(buf_id=0; buf_id<num_buf; buf_id++)
187  {
188  in_img[buf_id] = vxCreateImage(context, width, height, VX_DF_IMAGE_U8);
189  out_img[buf_id] = vxCreateImage(context, width, height, VX_DF_IMAGE_U8);
190  }
191  tmp = vxCreateImage(context, width, height, VX_DF_IMAGE_U8);
199  n0 = vxNotNode(graph, in_img[0], tmp);
200  n1 = vxNotNode(graph, tmp, out_img[0]);
201 
202  vxSetNodeTarget(n0, VX_TARGET_STRING, TIVX_TARGET_DSP1);
203  vxSetNodeTarget(n1, VX_TARGET_STRING, TIVX_TARGET_DSP2);
211  /* input @ n0 index 0, becomes graph parameter 0 */
212  add_graph_parameter_by_node_index(graph, n0, 0);
213  /* output @ n1 index 1, becomes graph parameter 1 */
214  add_graph_parameter_by_node_index(graph, n1, 1);
227  /* set graph schedule config such that graph parameter @ index 0 and 1 are enqueuable */
228  graph_parameters_queue_params_list[0].graph_parameter_index = 0;
229  graph_parameters_queue_params_list[0].refs_list_size = num_buf;
230  graph_parameters_queue_params_list[0].refs_list = (vx_reference*)&in_img[0];
231 
232  graph_parameters_queue_params_list[1].graph_parameter_index = 1;
233  graph_parameters_queue_params_list[1].refs_list_size = num_buf;
234  graph_parameters_queue_params_list[1].refs_list = (vx_reference*)&out_img[0];
235 
236  /* Schedule mode auto is used, here we dont need to call vxScheduleGraph
237  * Graph gets scheduled automatically as refs are enqueued to it
238  */
239  vxSetGraphScheduleConfig(graph,
240  VX_GRAPH_SCHEDULE_MODE_QUEUE_AUTO,
241  2,
242  graph_parameters_queue_params_list
243  );
251  /* explicitly set graph pipeline depth */
252  tivxSetGraphPipelineDepth(graph, pipeline_depth);
253 
254  /* make the 'tmp' reference which is output of n0 @ index 1 a "multi-buffer" */
255  tivxSetNodeParameterNumBufByIndex(n0, 1, num_buf);
265  vxVerifyGraph(graph);
266 
267  tivxExportGraphToDot(graph, ".", "vx_tutorial_graph_pipeline_two_nodes");
268  #if 0
269  tivxLogRtTrace(graph);
270  #endif
271 
273  exe_time = tivxPlatformGetTimeInUsecs();
274 
281  for(buf_id=0; buf_id<num_buf; buf_id++)
282  {
283  /* reset output */
284  vxGraphParameterEnqueueReadyRef(graph, 1, (vx_reference*)&out_img[buf_id], 1);
285  /* load input */
286  vxGraphParameterEnqueueReadyRef(graph, 0, (vx_reference*)&in_img[buf_id], 1);
287  }
290  buf_id = 0;
291 
297  for(loop_id=0; loop_id<(loop_cnt+num_buf); loop_id++)
298  {
299  vx_image cur_out_img, cur_in_img;
300  uint32_t num_refs;
301 
302  /* Get output reference, waits until a reference is available */
303  vxGraphParameterDequeueDoneRef(graph, 1, (vx_reference*)&cur_out_img, 1, &num_refs);
304 
305  /* Get consumed input reference, waits until a reference is available
306  */
307  vxGraphParameterDequeueDoneRef(graph, 0, (vx_reference*)&cur_in_img, 1, &num_refs);
308 
309  /* A graph execution completed, since we dequeued both input and output refs */
310 
311  /* use output */
312 
313  buf_id = (buf_id+1)%num_buf;
314 
315  /* recycles dequeued input and output refs 'loop_cnt' times */
316  if(loop_id<loop_cnt)
317  {
318  /* input and output can be enqueued in any order */
319  vxGraphParameterEnqueueReadyRef(graph, 1, (vx_reference*)&cur_out_img, 1);
320  vxGraphParameterEnqueueReadyRef(graph, 0, (vx_reference*)&cur_in_img, 1);
321  }
322  }
323  /* ensure all graph processing is complete */
324  vxWaitGraph(graph);
327  exe_time = tivxPlatformGetTimeInUsecs() - exe_time;
328 
329  {
330  vx_node nodes[] = { n0, n1 };
331 
332  print_graph_pipeline_performance(graph, nodes, 2, exe_time, loop_cnt+num_buf, width*height);
333  }
334 
339  vxReleaseNode(&n0);
340  vxReleaseNode(&n1);
341  for(buf_id=0; buf_id<num_buf; buf_id++)
342  {
343  vxReleaseImage(&in_img[buf_id]);
344  vxReleaseImage(&out_img[buf_id]);
345  }
346  vxReleaseImage(&tmp);
347  vxReleaseGraph(&graph);
348  vxReleaseContext(&context);
351  printf(" vx_tutorial_graph_pipeline_two_nodes: Tutorial Done !!! \n");
352  printf(" \n");
353 }
