152  vx_graph graph = 0;

211

230 
231  if(VX_SUCCESS == status) {
232

235  VX_TUTORIAL_ASSERT(status==VX_SUCCESS);
236 
237  if (status!=VX_SUCCESS) {
238  goto exit;
239  }
240 
241  printf(" Execute graph ... \n");
242  /* Execute the network */
243  status = vxProcessGraph(graph);
244  VX_TUTORIAL_ASSERT(status==VX_SUCCESS);
245 
246  /* Display the output */
247  displayOutput(NULL, (vx_df_image)NULL, NULL, 0, 0, 0, config, &output_tensor, &obj->output_file_path[0], obj->operation_mode);
248 
249  }

252 
253  vxReleaseNode(&node);
254  vxReleaseGraph(&graph);
255 
256  vxReleaseUserDataObject(&config);
257 
258  vxReleaseUserDataObject(&network);
259  vxReleaseTensor(&input_tensor);
260  vxReleaseTensor(&output_tensor);
261 
262  vxRemoveKernel(kernel);
263 
264  printf("\n---- Execution time: %4.6f ms\n", perf_graph.min/1000000.0);
265 
266  exit:
267  printf("\n vx_tutorial_tidl: Tutorial Done !!! \n");
268  printf(" \n");
269 
270  vxReleaseContext(&context);
271

273  void tivxSetSelfCpuId(vx_enum cpu_id);

275 #endif

279 static vx_status parse_cfg_file(VxTutorialTidl_CfgObj *obj, char *cfg_file_name)
280 {
281  FILE *fp = fopen(cfg_file_name, "r");
282  char line_str[1024];
283  char *token;
284  size_t sizeFilePath;
285  char filePath[MAXPATHLENGTH];
286  vx_status status = VX_SUCCESS;
287 
288  if(fp==NULL)
289  {
290  printf("# ERROR: Unable to open config file [%s]\n", cfg_file_name);
291 #ifdef HOST_EMULATION
292  printf("# ERROR: Please make sure that the environment variable VX_TEST_DATA_PATH is set to .../conformance_tests/test_data\n");
293 #endif
294  status= VX_FAILURE;
295  goto exit;
296  }
297 
298  while(fgets(line_str, sizeof(line_str), fp)!=NULL)
299  {
300  char s[]=" \t";
301 
302  if (strchr(line_str, '#'))
303  {
304  continue;
305  }
306 
307  /* get the first token */
308  token = strtok(line_str, s);
309 
310  if(strcmp(token, "tidl_params_file_path")==0)
311  {
312  token = strtok(NULL, s);
313  token[strlen(token)-1]=0;
314  sizeFilePath = snprintf(filePath, MAXPATHLENGTH, "%s/%s", ct_get_test_file_path(), token);
315  if (sizeFilePath > MAXPATHLENGTH) {
316  printf("Error in parse_cfg_file, path too long to fit in string\n");
317  }
318  else {
319  strcpy(obj->tidl_params_file_path, filePath);
320  }
321  }
322  else
323  if(strcmp(token, "tidl_network_file_path")==0)
324  {
325  token = strtok(NULL, s);
326  token[strlen(token)-1]=0;
327  sizeFilePath = snprintf(filePath, MAXPATHLENGTH, "%s/%s", ct_get_test_file_path(), token);
328  if (sizeFilePath > MAXPATHLENGTH) {
329  printf("Error in parse_cfg_file, path too long to fit in string\n");
330  }
331  else {
332  strcpy(obj->tidl_network_file_path, filePath);
333  }
334  }
335  else
336  if(strcmp(token, "input_file_path")==0)
337  {
338  token = strtok(NULL, s);
339  token[strlen(token)-1]=0;
340  sizeFilePath = snprintf(filePath, MAXPATHLENGTH, "%s/%s", ct_get_test_file_path(), token);
341  if (sizeFilePath > MAXPATHLENGTH) {
342  printf("Error in parse_cfg_file, path too long to fit in string\n");
343  }
344  else {
345  strcpy(obj->input_file_path, filePath);
346  }
347  }
348  else
349  if(strcmp(token, "output_file_path")==0)
350  {
351  token = strtok(NULL, s);
352  token[strlen(token)-1]=0;
353  sizeFilePath = snprintf(filePath, MAXPATHLENGTH, "%s/%s", ct_get_test_file_path(), token);
354  if (sizeFilePath > MAXPATHLENGTH) {
355  printf("Error in parse_cfg_file, path too long to fit in string\n");
356  }
357  else {
358  strcpy(obj->output_file_path, filePath);
359  }
360  }
361  else
362  if(strcmp(token, "operation_mode")==0)
363  {
364  token = strtok(NULL, s);
365  obj->operation_mode = atoi(token);
366  }
367  }
368 
369  fclose(fp);
370 
371  exit:
372  return status;
373 }

376 static vx_tensor createInputTensor(vx_context context, vx_user_data_object config)
377 {
378  vx_size input_sizes[VX_TUTORIAL_MAX_TENSOR_DIMS];
379  vx_map_id map_id_config;
380  sTIDL_IOBufDesc_t *ioBufDesc;
381 
382  vxMapUserDataObject(config, 0, sizeof(sTIDL_IOBufDesc_t), &map_id_config,
383  (void **)&ioBufDesc, VX_READ_ONLY, VX_MEMORY_TYPE_HOST, 0);
384 
385  input_sizes[0] = ioBufDesc->inWidth[0] + ioBufDesc->inPadL[0] + ioBufDesc->inPadR[0];
386  input_sizes[1] = ioBufDesc->inHeight[0] + ioBufDesc->inPadT[0] + ioBufDesc->inPadB[0];
387  input_sizes[2] = ioBufDesc->inNumChannels[0];
388 
389  vxUnmapUserDataObject(config, map_id_config);
390 
391  return vxCreateTensor(context, 3, input_sizes, VX_TYPE_UINT8, 0);
392 }

414 static vx_status readDataS8(FILE *fp, int8_t *ptr, int32_t n,
415  int32_t width, int32_t height, int32_t pitch,
416  int32_t chOffset)
417 {
418  int32_t i0, i1;
419  uint32_t readSize;
420  vx_status status = VX_SUCCESS;
421 
422  for(i0 = 0; i0 < n; i0++)
423  {
424  for(i1 = 0; i1 < height; i1++)
425  {
426  readSize= fread(&ptr[i0*chOffset + i1*pitch], 1, width, fp);
427  if (readSize != width) {
428  status= VX_FAILURE;
429  goto exit;
430  }
431  }
432  }
433 
434  exit:
435  return status;

440 static vx_status readInput(vx_context context, vx_user_data_object config, vx_tensor *input_tensors, char *input_file, uint32_t operation_mode)
441 {
442  vx_status status = VX_SUCCESS;
443 
444  int8_t *input_buffer = NULL;
445  uint32_t id;
446 
447  vx_map_id map_id_config;
448  vx_map_id map_id_input;
449 
450  vx_size start[VX_TUTORIAL_MAX_TENSOR_DIMS];
451  vx_size input_strides[VX_TUTORIAL_MAX_TENSOR_DIMS];
452  vx_size input_sizes[VX_TUTORIAL_MAX_TENSOR_DIMS];
453 
454  sTIDL_IOBufDesc_t *ioBufDesc;
455 
456  FILE *fp;
457 
458  fp= fopen(input_file, "rb");
459 
460  if(fp==NULL)
461  {
462  printf("# ERROR: Unable to open input file [%s]\n", input_file);
463  return(VX_FAILURE);
464  }
465 
466  vxMapUserDataObject(config, 0, sizeof(sTIDL_IOBufDesc_t), &map_id_config,
467  (void **)&ioBufDesc, VX_READ_ONLY, VX_MEMORY_TYPE_HOST, 0);
468 
469  for(id = 0; id < ioBufDesc->numInputBuf; id++)
470  {
471  input_sizes[0] = ioBufDesc->inWidth[id] + ioBufDesc->inPadL[id] + ioBufDesc->inPadR[id];
472  input_sizes[1] = ioBufDesc->inHeight[id] + ioBufDesc->inPadT[id] + ioBufDesc->inPadB[id];
473  input_sizes[2] = ioBufDesc->inNumChannels[id];
474 
475  start[0] = start[1] = start[2] = 0;
476 
477  input_strides[0] = 1;
478  input_strides[1] = input_sizes[0];
479  input_strides[2] = input_sizes[1] * input_strides[1];
480 
481  status = tivxMapTensorPatch(input_tensors[id], 3, start, input_sizes, &map_id_input, input_strides, (void **)&input_buffer, VX_WRITE_ONLY, VX_MEMORY_TYPE_HOST);
482 
483  if (VX_SUCCESS == status)
484  {
485  status= readDataS8(
486  fp,
487  &input_buffer[(ioBufDesc->inPadT[id] * input_strides[1]) + ioBufDesc->inPadL[id]],
488  ioBufDesc->inNumChannels[id],
489  ioBufDesc->inWidth[id],
490  ioBufDesc->inHeight[id],
491  input_strides[1],
492  input_strides[2]);
493 
494  tivxUnmapTensorPatch(input_tensors[id], map_id_input);
495 
496  if (status== VX_FAILURE) {
497  goto exit;
498  }
499  }
500  }
501 
502  exit:
503  vxUnmapUserDataObject(config, map_id_config);
504 
505  fclose(fp);
506 
507  return status;
508 }

510 static void displayOutput(void *bmp_context, vx_df_image df_image, void *data_ptr, vx_uint32 img_width, vx_uint32 img_height, vx_uint32 img_stride, vx_user_data_object config, vx_tensor *output_tensors, char *output_file, uint32_t operation_mode)
511 {
512  vx_status status = VX_SUCCESS;
513 
514  vx_size output_sizes[VX_TUTORIAL_MAX_TENSOR_DIMS];
515 
516  vx_map_id map_id_config;
517 
518  int32_t id, i, j;
519 
520  sTIDL_IOBufDesc_t *ioBufDesc;
521 
522  vxMapUserDataObject(config, 0, sizeof(sTIDL_IOBufDesc_t), &map_id_config,
523  (void **)&ioBufDesc, VX_READ_ONLY, VX_MEMORY_TYPE_HOST, 0);
524 
525  for(id = 0; id < ioBufDesc->numOutputBuf; id++)
526  {
527  output_sizes[0] = ioBufDesc->outWidth[id] + ioBufDesc->outPadL[id] + ioBufDesc->outPadR[id];
528  output_sizes[1] = ioBufDesc->outHeight[id] + ioBufDesc->outPadT[id] + ioBufDesc->outPadB[id];
529  output_sizes[2] = ioBufDesc->outNumChannels[id];
530 
531  status = vxGetStatus((vx_reference)output_tensors[id]);
532 
533  if (VX_SUCCESS == status)
534  {
535  void *output_buffer;
536 
537  vx_map_id map_id_output;
538 
539  vx_size output_strides[VX_TUTORIAL_MAX_TENSOR_DIMS];
540  vx_size start[VX_TUTORIAL_MAX_TENSOR_DIMS];
541 
542  start[0] = start[1] = start[2] = start[3] = 0;
543 
544  output_strides[0] = 1;
545  output_strides[1] = output_sizes[0];
546  output_strides[2] = output_sizes[1] * output_strides[1];
547 
548  tivxMapTensorPatch(output_tensors[id], 3, start, output_sizes, &map_id_output, output_strides, &output_buffer, VX_READ_ONLY, VX_MEMORY_TYPE_HOST);
549 
550  if (operation_mode == 0)
551  {
552  uint8_t *pOut;
553  uint8_t score[5];
554  vx_uint32 classid[5];
555 
556  pOut = (uint8_t *)output_buffer + (ioBufDesc->outPadT[id] * output_sizes[0]) + ioBufDesc->outPadL[id];
557 
558  for(i = 0; i < 5; i++)
559  {
560  score[i] = 0;
561  classid[i] = 0xFFFFFFFF;
562 
563  for(j = 0; j < output_sizes[0]; j++)
564  {
565  if(pOut[j] > score[i])
566  {
567  score[i] = pOut[j];
568  classid[i] = j;
569  }
570  }
571 
572  pOut[classid[i]] = 0;
573  }
574 
575  printf("\nImage classification Top-5 results: \n");
576 
577  for(i = 0; i < 5; i++)
578  {
579  printf(" %s, class-id: %d, score: %u\n", (char *)&imgnet_labels[classid[i]], classid[i], score[i]);
580  }
581  }
582  else
583  if (operation_mode == 1)
584  {
585  /* Display of detected objects not supported for now */
586  }
587  tivxUnmapTensorPatch(output_tensors[id], map_id_output);
588  }
589  }
590 
591  vxUnmapUserDataObject(config, map_id_config);
592 }
