195  vx_uint32 x, vx_uint32 y, const vx_imagepatch_addressing_t *addr)
196 {
197  return (addr->stride_y * (y / addr->step_y)) +
198  (addr->stride_x * (x / addr->step_x));

204 static inline vx_status tivxCheckNullParams(
205  tivx_obj_desc_t *obj_desc[], uint16_t num_params,
206  uint16_t max_params)
207 {
208  vx_status status = VX_SUCCESS;
209  uint32_t i;
210 
211  if (num_params != max_params)
212  {
213  status = VX_FAILURE;
214  }
215  else
216  {
217  for (i = 0U; i < max_params; i ++)
218  {
219  if (NULL == obj_desc[i])
220  {
221  status = VX_FAILURE;
222  break;
223  }
224  }
225  }
226  return status;
227 }
